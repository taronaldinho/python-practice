# [じゃんけんゲームを作ってみる](じゃんけんゲームを作ってみる.md)

```Python
# まずは基本的なじゃんけんの仕組みを実装する。
print("=" * 60)

# 人間の入力を求める。
human_shape = input(
    "じゃんけんの「手」をキーボードで入力してください。\n"
    "\tR/r: Rock（グー）\n"
    "\tP/p: Paper（パー）\n"
    "\tS/s: Scissors（チョキ）\n\n"
).upper()  # 入力はすべて大文字に変換しておく。

print("- " * 30)

# 勝ち負け判定を行う。コンピュータはR（グー）を出すものとする。
result = "不適切な入力"
if human_shape == "R":
    result = "あいこ"
elif human_shape == "P":
    result = "あなたの勝ち"
elif human_shape == "S":
    result = "あなたの負け"

# 結果を表示させる。
print(f"あなたの手: {human_shape}\tコンピュータの手: R\t【結果】{result}です。\n")
```

```Python
import random

    # 人間の入力を求める。

        "\tQ/q: Quit（やめる）\n"
    ).upper()  # 入力はすべて大文字に変換しておく。

    print("- " * 30)

    # プログラムを終了するかどうかチェックする。
    if human_shape == "Q":
        print("じゃんけんを終了します。")
        quit()  # プログラムを終了する。

    # コンピュータの手をランダムに選択させる。
    cpu_shape = random.choice(["R", "P", "S"])

    # 勝ち負け判定を行う。
    result = "不適切な入力"
    if human_shape == cpu_shape:
        result = "あいこ"
    elif human_shape == "R":
        if cpu_shape == "P":
            result = "あなたの負け"
        else:
            result = "あなたの勝ち"
    elif human_shape == "P":
        if cpu_shape == "S":
            result = "あなたの負け"
        else:
            result = "あなたの勝ち"
    elif human_shape == "S":
        if cpu_shape == "R":
            result = "あなたの負け"
        else:
            result = "あなたの勝ち"

    # 結果を表示させる。
    print(f"あなたの手: {human_shape}\tコンピュータの手: {cpu_shape}\t【結果】{result}です。\n")
```

```Python
# 勝ち負けの判定に数理的なアルゴリズムを導入する。
# Python標準のrandomモジュールを利用する。
import random

# 有効な入力と対応する値の一覧。
valid_inputs = {"R": 0, "P": 1, "S": 2, "Q": -1}  # "Q" に対応する値-1は使用しない。

# 勝利数を初期化する。
human_wins = 0
cpu_wins = 0
draws = 0

# 無限ループして続けて何度もじゃんけんできるようにする。
while True:
    print("=" * 60)

    # 人間の入力を求める。
    human_shape = input(
        "じゃんけんの「手」をキーボードで入力してください。\n"
        "\tR/r: Rock（グー）\n"
        "\tP/p: Paper（パー）\n"
        "\tS/s: Scissors（チョキ）\n\n"
        "\tQ/q: Quit（やめる）\n"
    ).upper()  # 入力はすべて大文字に変換しておく。

    print("- " * 30)

    # 有効な入力かどうか、プログラムを終了するかどうかのチェック。
    if human_shape not in valid_inputs.keys():
        print(f"適切な文字を入力してください（{valid_inputs.keys()}）。\n")
        continue  # ループの先頭に戻る。
    elif human_shape == "Q":
        print("じゃんけんを終了します。")
        quit()  # プログラムを終了する。

    # コンピュータの手をランダムに選択させる。
    cpu_shape = random.choice(["R", "P", "S"])

    # 勝ち負け判定アルゴリズム。
    result = "あなたの勝ち"
    if human_shape == cpu_shape:
        result = "あいこ"
        draws += 1
    elif (valid_inputs[human_shape] - valid_inputs[cpu_shape] + 3) % 3 == 2:
        result = "あなたの負け"
        cpu_wins += 1
    else:
        human_wins += 1

    # 結果を表示させる。
    print(f"あなたの手: {human_shape}\tコンピュータの手: {cpu_shape}\t【結果】{result}です。\n")
    print(f"あなたの勝利回数: {human_wins}\tコンピュータの勝利回数: {cpu_wins}\tあいこの回数: {draws}\n")
```
